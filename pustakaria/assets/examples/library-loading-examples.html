Basic Usage

// Auto-initialize (already included in layout) // Loading will show automatically on page load // Manual control showLibraryLoading('Memuat data...'); hideLibraryLoading();

Custom Text

// Update loading text updateLibraryLoadingText('Perpustakaan Digital', 'Mengunduh koleksi...'); // Show with custom subtitle showLibraryLoading('Menyimpan perubahan...');

Form Integration

// Automatic loading on form submit <form action="/save" method="post"> <!-- Loading will show automatically --> <button type="submit">Simpan</button> </form> // Disable auto-loading for specific form <form action="/search" method="get" data-no-loading> <!-- No loading will show --> <button type="submit">Cari</button> </form>

Advanced Configuration

// Custom initialization const customLoading = new LibraryLoading({ minDisplayTime: 2000, // Minimum 2 seconds fadeOutDuration: 800, // 800ms fade out autoHide: false, // Manual control showOnNavigation: true // Show on page navigation }); // Show with custom options customLoading.show('Memproses...'); setTimeout(() => customLoading.hide(), 5000);
